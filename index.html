-- Table 1: Employee profiles create table if not exists profiles ( employee_id text primary key, full_name text not null, current_state text not null default 'OUT' check (current_state in ('OUT','IN','ON_BREAK')) ); -- Table 2: Activity log (server timestamp auto-applied via default) create table if not exists time_logs ( id bigserial primary key, employee_id text not null references profiles(employee_id) on delete cascade, action_type text not null, logged_at timestamptz not null default now() ); -- Enable Row Level Security alter table profiles enable row level security; alter table time_logs enable row level security; -- Allow anonymous employee ID lookup (single-row, filtered in app layer) create policy "employee_self_lookup" on profiles for select using (true); -- Allow time log inserts from the app create policy "allow_log_insert" on time_logs for insert with check (true); -- Allow state updates (clock in/out/break) create policy "allow_state_update" on profiles for update using (true); -- Allow admin to add new employee profiles create policy "allow_profile_insert" on profiles for insert with check (true);
